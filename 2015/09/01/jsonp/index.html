使用jsonp跨域调用百度js实现的智能提示框

JSONP原理

JSONP是JSON width padding（填充式或参数式JSON）的简写，是应用JSON的一种新方法，在后来的Web服务中非常流行。
JSONP看起来和JSON差不多，只不过是被包含在函数调用中的HSON，就像下面这样

callback({"name":"Nicholas"});

JSONP有两部分组成：回调函数和数据。回调函数是当响应到来时应该在页面中调用的函数。回调函数的名字一般是在请
求中指定的。而数据就是传入回调函数的JSON数据。下面是一个典型的JSONP请求。

http://freegeoip.net/?callback=handleResponse

这个URL是在请求一个JSONP地理定位服务。通过查询字符串来指定JSONP服务的回调函数参数是很常见的，就像上面的URL
所示，这里指定指定的回调函数名字叫handleResponse()。

JSONP是通过动态 < script >元素来使用的，使用时可以为src属性指定一个跨域URL。这里的< script >元素与< img >元素类
似，都有能力不受限制地从其它域加载资源。因为JSONP是有效的JavaScript代码，所以在请求完成后，即在JSONP响应加载到
页面中以后，就会立即执行。来看一个例子。

function handleResponse(response) {
    alert("You're at IP address " + response.ip + ",which is in " + response.city + ", " + 
    response.region_name);
}

var script = document.createElement("script");
script.src = "http://freegeoip.net/json/?callback=handleResponse";
document.body.insertBefore(script,document.body.firstChild);

这个例子通过查询地理位置定位服务来显示你IP地址和位置信息。我们要获取的是http://freegeoip.net 这个域中的数据，以上代码
是本地的JS文件，当动态的添加< script >后，在另一个域http://freegeoip.net 中则会生成这样一个回调函数（原理是字符串的获
取，这属于后台的范畴）

handleResponse({
    ip: 59.44.121.218,
    city: "沈阳",
    region_name："辽宁"
});

我们不会看到这个运行函数，因为它在http://freegeoip.net 中，我们只需要动态添加< script >调用就行,前端所需要做的是知道这
个域有这些数据，在本地完成第一段代码进行调用即可，这样就能完成整个过程，有效弹出信息。

<input type="text" id="txtSuggestions" />
<script type="text/javascript">

var txt = document.getElementById("txtSuggestions");
txt.onkeyup = function() {
    var inputTxt = txt.value;
    var script = document.createElement("script");
        script.src = "https://sp0.baidu.com/5a1Fazu8AA54nxGko9WTAnF6hhy/su/json/?wd="+inputTxt+"&p=3&cb=getData&t=new Date().getMilliseconds().toString()";
        document.body.insertBefore(script,document.body.firstChild);
    };
function getData(data) {
    var Info = data.s; //获取异步数据,可以先输出data即可知道我们需要数据在data.s里
    console.log(Info);
}
</script>

使用jquery和ajax完成

<script type="text/javascript">
$(function () {
    $("#txtSuggestions").keyup(function () {
        var inputTxt = $("#txtSuggestions").val();   //获取输入的关键字
        var qsData = { 'wd': inputTxt, 'p': '3', 'cb': 'getData', 't': new Date().getMilliseconds().toString() };
        $.ajax({
            async: false,
            url: "https://sp0.baidu.com/5a1Fazu8AA54nxGko9WTAnF6hhy/su",
            type: "GET",
            dataType: 'jsonp',
            jsonp: 'jsoncallback',
            data: qsData,
            timeout: 5000,
            success: function (json) {
            },
            error: function (xhr) {
            }
        });
    });
});
function getData(data) {
    var Info = data.s; 
    console.log(Info);
}
</script>
